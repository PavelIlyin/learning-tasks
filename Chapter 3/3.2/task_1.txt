function pow(x,n)			// нет пробела между аргументами (после запятой)
{					// фигурная скобка на отдельной строке, нужно поднять выше
  let result=1;				// нет пробела справа и слева от знака =
  for(let i=0;i<n;i++) {result*=x;}	// блок кода разделить на две строки (вынести ниже), нет пробелов
  return result;
}

let x=prompt("x?",''), n=prompt("n?",'') // лучше разделить на две строки, нет пробелов, нет точик с запятой
if (n<=0)                                // нет пробелов справа и слева от знака =
{					 // фигурная скобка на отдельной строке, нужно поднять выше	
  alert(`Степень ${n} не поддерживается, введите целую степень, большую 0`); // строку разбить для читаемости
}					 
else                                     // строку можно поставить перед верхней фигурной скобкой
{					 // фигурную скобку поставить перед else
  alert(pow(x,n))			 // нет пробелов
}
